1- Quelques fonctions

Fonctions prédéfinies
Fonctions prédéfinies Description
SELECT CURDATE() Affiche la date courante : quand on aura besoin de la date du jour dans la requête
SELECT CURDATE() +0 Affiche la date courante sans tirets, pratique pour nommer un fichier
SELECT CURTIME() Affiche l'heure
SELECT NOW Affiche la date et l'heure
SELECT DATE-ADD('2021-12-06', INTERVAL 31 DAY) Affiche la date saisie et celle qui correspond aux jours, mois, années ajoutés, ici: '31 DAY'
SELECT DATE_FORMAT('2012-10-03 22:23:00','%d/%m/%Y - %H/%i/%s') Change le format de la date du jour au format FR
SELECT date_rendu, DATE_FORMAT(date_rendu, '%d/%m/%Y')AS'date rendue en fr' FROM emprunt Exemple sur la BDD bibliothèque
SELECT CONCAT_WS('-','a','b','c') WS with separator utile pour concatener(ajouter) un code postale avec une ville
SELECT TRIM('coucou') Supprime les espaces en début et en fin de chaîne
